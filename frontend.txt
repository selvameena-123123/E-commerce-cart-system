<script>
// CART
let cart=[];
function addToCart(name,price,img){
  cart.push({name,price,img});
  updateCart();
}
function removeFromCart(index){
  cart.splice(index,1);
  updateCart();
}
function updateCart(){
  const cartItems=document.getElementById("cart-items");
  const cartTotal=document.getElementById("cart-total");
  const cartCount=document.getElementById("cart-count");
  cartItems.innerHTML="";
  let total=0;
  cart.forEach((item,i)=>{
    total+=item.price;
    cartItems.innerHTML+=`
      <div class="cart-item">
        <img src="${item.img}" />
        <span>${item.name}</span>
        <span>‚Çπ${item.price}</span>
        <button class="remove-btn" onclick="removeFromCart(${i})">x</button>
      </div>`;
  });
  cartTotal.innerText="Total: ‚Çπ"+total;
  cartCount.innerText=cart.length;
}
function toggleCart(){document.getElementById("cart").classList.toggle("open");}

// LOGIN
function openLogin(){document.getElementById('loginModal').style.display='flex';}
function closeLogin(){document.getElementById('loginModal').style.display='none';}

// LOAD PRODUCTS FROM API (Multiple pages for more products)
let allProducts=[];
async function fetchProducts(){
  for(let skip=0; skip<200; skip+=100){
    let res=await fetch(https://dummyjson.com/products?limit=100&skip=${skip});
    let data=await res.json();
    allProducts.push(...data.products);
  }
  displayProducts(allProducts);
  loadCategories(allProducts);
}
fetchProducts();

function displayProducts(products){
  const container=document.getElementById('productContainer');
  container.innerHTML="";
  products.forEach(p=>{
    container.innerHTML+=`
      <div class="card ${p.category}">
        <img src="${p.thumbnail}" />
        <h3>${p.title}</h3>
        <p>‚Çπ${p.price}</p>
        <button onclick="addToCart('${p.title}',${p.price},'${p.thumbnail}')">Add to Cart</button>
      </div>`;
  });
}

// CATEGORIES
function loadCategories(products){
  const categories=[...new Set(products.map(p=>p.category))];
  const sidebar=document.getElementById('categorySidebar');
  sidebar.innerHTML=<button onclick="filterProducts('all')">All</button>;
  categories.forEach(cat=>{
    sidebar.innerHTML+=<button onclick="filterProducts('${cat}')">${cat.charAt(0).toUpperCase()+cat.slice(1)}</button>;
  });
}

// FILTER
function filterProducts(category){
  if(category==='all') displayProducts(allProducts);
  else displayProducts(allProducts.filter(p=>p.category===category));
}

// SEARCH
function searchProducts(){
  const query=document.getElementById('searchInput').value.toLowerCase();
  const filtered=allProducts.filter(p=>p.title.toLowerCase().includes(query));
  displayProducts(filtered);
}
